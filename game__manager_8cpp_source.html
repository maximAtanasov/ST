80 void game_manager::handle_messages(){
81  auto temp = msg_sub.get_next_message();
82  while(temp != nullptr){
83  switch (temp->msg_name) {
84  case LOAD_LEVEL:
85  load_level(*static_cast<std::string*>(temp->get_data()));
86  break;
87  case RELOAD_LEVEL:
88  reload_level(*static_cast<std::string*>(temp->get_data()));
89  break;
90  case START_LEVEL:
91  start_level(*static_cast<std::string*>(temp->get_data()));
92  break;
93  case UNLOAD_LEVEL:
94  unload_level(*static_cast<std::string*>(temp->get_data()));
95  break;
96  case KEY_PRESSED: {
97  uint8_t key_index = temp->base_data0;
98  keys_pressed_data[key_index] = true;
99  keys_held_data[key_index] = false;
100  keys_released_data[key_index] = false;
101  break;
102  }
103  case KEY_HELD: {
104  uint8_t key_index = temp->base_data0;
105  keys_pressed_data[key_index] = false;
106  keys_held_data[key_index] = true;
107  keys_released_data[key_index] = false;
108  break;
109  }
110  case KEY_RELEASED: {
111  uint8_t key_index = temp->base_data0;
112  keys_pressed_data[key_index] = false;
113  keys_held_data[key_index] = false;
114  keys_released_data[key_index] = true;
115  break;
116  }
117  case MOUSE_X:
118  mouse_x = static_cast<int32_t>(temp->base_data0);
119  break;
120  case MOUSE_Y:
121  mouse_y = static_cast<int32_t>(temp->base_data0);
122  break;
123  case LEFT_TRIGGER:
124  left_trigger = static_cast<int16_t>(temp->base_data0);
125  break;
126  case RIGHT_TRIGGER:
127  right_trigger = static_cast<int16_t>(temp->base_data0);
128  break;
129  case LEFT_STICK_VERTICAL:
130  left_stick_vertical = static_cast<int16_t>(temp->base_data0);
131  break;
132  case LEFT_STICK_HORIZONTAL:
133  left_stick_horizontal = static_cast<int16_t>(temp->base_data0);
134  break;
135  case RIGHT_STICK_VERTICAL:
136  right_stick_vertical = static_cast<int16_t>(temp->base_data0);
137  break;
138  case RIGHT_STICK_HORIZONTAL:
139  right_stick_horizontal = static_cast<int16_t>(temp->base_data0);
140  break;
141  case MUSIC_VOLUME_LEVEL:
142  music_volume_level = static_cast<uint8_t>(temp->base_data0);
143  break;
144  case SOUNDS_VOLUME_LEVEL:
145  sounds_volume_level = static_cast<uint8_t>(temp->base_data0);
146  break;
147  case AUDIO_ENABLED:
148  audio_enabled = static_cast<bool>(temp->base_data0);
149  break;
150  case VSYNC_STATE:
151  vsync_flag = static_cast<bool>(temp->base_data0);
152  break;
153  case END_GAME:
154  game_is_running_ = false;
155  break;
156  case SHOW_MOUSE:
157  SDL_ShowCursor(static_cast<bool>(temp->base_data0));
158  break;
159  case EXECUTE_SCRIPT: {
160  auto script = static_cast<std::string*>(temp->get_data());
161  gScript_backend.run_script(*script);
162  break;
163  }
164  case FULLSCREEN_STATUS:
165  fullscreen_status = static_cast<bool>(temp->base_data0);
166  break;
167  case VIRTUAL_SCREEN_COORDINATES: {
168  auto data = temp->base_data0;
169  v_width = data & 0x0000ffffU;
170  v_height = (data >> 16U) & 0x0000ffffU;
171  break;
172  }
173  }
174  delete temp;
175  temp = msg_sub.get_next_message();
176  }
177 }

183 int8_t game_manager::load_level(const std::string& level_name){
184 
185  //check if it is already loaded.
186  for(auto& i: levels) {
187  if (i.get_name() == level_name) {
188  return 0;
189  }
190  }
191 
192  //otherwise - create it
193  auto temp = ST::level(level_name, &gMessage_bus);
194  if(temp.load() != 0){
195  gMessage_bus.send_msg(new message(LOG_ERROR, make_data<std::string>("Level with name " + level_name + " could not be loaded!")));
196  return -1;
197  }
198  levels.emplace_back(temp);
199 
200  //current level pointer must be reset, because apparently adding to the vector changes the pointer address
201  //(makes sense as it has to reallocate the whole thing)
202  for (auto &level : levels) {
203  if(level.get_name() == active_level){
204  current_level_pointer = &level;
205  break;
206  }
207  }
208  return 0;
209 }

215 void game_manager::unload_level(const std::string& level_name){
216  for(uint64_t i = 0; i < levels.size(); ++i) {
217  if (levels[i].get_name() == level_name) {
218  levels[i].unload();
219  levels.erase(levels.begin() + i);
220  break;
221  }
222  }
223 
224  //current level pointer must be reset, because apparently adding/removing to/from the vector changes the pointer address
225  //(makes sense as it has to reallocate the whole thing)
226  for (auto &level : levels) {
227  if(level.get_name() == active_level) {
228  current_level_pointer = &level;
229  break;
230  }
231  }
232 }

253 void game_manager::start_level(const std::string& level_name) {
254 
255  //set the current level pointer
256  bool is_loaded = false;
257  for (auto &level : levels) {
258  if(level.get_name() == level_name) {
259  current_level_pointer = &level;
260  is_loaded = true;
261  break;
262  }
263  }
264  //if the level wasn't loaded in advance, load it now
265  if(!is_loaded){
266  if(load_level(level_name) == 0) {
267  for (auto &level : levels) {
268  if (level.get_name() == level_name) {
269  current_level_pointer = &level;
270  break;
271  }
272  }
273  } else {
274  gMessage_bus.send_msg(new message(LOG_ERROR, make_data<std::string>("Error starting level " + level_name)));
275  return;
276  }
277  }
278 
279  gScript_backend.close();
280  gScript_backend.initialize(&gMessage_bus, this);
281  active_level = level_name;
282 
283  get_level()->lights.clear();
284  get_level()->entities.clear();
285  get_level()->text_objects.clear();
286 
287  //construct level
288  get_level()->camera.x = 0;
289  get_level()->camera.y = 0;
290 
291  std::string temp = "levels/";
292  temp = temp + active_level;
293  temp = temp + "/level.lua";
294  gScript_backend.run_file(temp);
295 }
