79 void game_manager::handle_messages(){
80  auto temp = msg_sub.get_next_message();
81  while(temp != nullptr){
82  if(temp->msg_name == LOAD_LEVEL){
83  load_level(*static_cast<std::string*>(temp->get_data()));
84  }
85  else if(temp->msg_name == RELOAD_LEVEL){
86  reload_level(*static_cast<std::string*>(temp->get_data()));
87  }
88  else if(temp->msg_name == START_LEVEL){
89  start_level(*static_cast<std::string*>(temp->get_data()));
90  }
91  else if(temp->msg_name == UNLOAD_LEVEL){
92  unload_level(*static_cast<std::string*>(temp->get_data()));
93  }
94  else if(temp->msg_name == KEY_PRESSED){
95  uint8_t key_index = temp->base_data0;
96  keys_pressed_data[key_index] = true;
97  keys_held_data[key_index] = false;
98  keys_released_data[key_index] = false;
99  }
100  else if(temp->msg_name == KEY_HELD){
101  uint8_t key_index = temp->base_data0;
102  keys_pressed_data[key_index] = false;
103  keys_held_data[key_index] = true;
104  keys_released_data[key_index] = false;
105  }
106  else if(temp->msg_name == KEY_RELEASED){
107  uint8_t key_index = temp->base_data0;
108  keys_pressed_data[key_index] = false;
109  keys_held_data[key_index] = false;
110  keys_released_data[key_index] = true;
111  }
112  else if(temp->msg_name == MOUSE_X){
113  mouse_x = static_cast<int32_t>(temp->base_data0);
114  }
115  else if(temp->msg_name == MOUSE_Y){
116  mouse_y = static_cast<int32_t>(temp->base_data0);
117  }
118  else if(temp->msg_name == LEFT_TRIGGER){
119  left_trigger = static_cast<int16_t>(temp->base_data0);
120  }
121  else if(temp->msg_name == RIGHT_TRIGGER){
122  right_trigger = static_cast<int16_t>(temp->base_data0);
123  }
124  else if(temp->msg_name == LEFT_STICK_VERTICAL){
125  left_stick_vertical = static_cast<int16_t>(temp->base_data0);
126  }
127  else if(temp->msg_name == LEFT_STICK_HORIZONTAL){
128  left_stick_horizontal = static_cast<int16_t>(temp->base_data0);
129  }
130  else if(temp->msg_name == RIGHT_STICK_VERTICAL){
131  right_stick_vertical = static_cast<int16_t>(temp->base_data0);
132  }
133  else if(temp->msg_name == RIGHT_STICK_HORIZONTAL){
134  right_stick_horizontal = static_cast<int16_t>(temp->base_data0);
135  }
136  else if(temp->msg_name == MUSIC_VOLUME_LEVEL){
137  music_volume_level = static_cast<uint8_t>(temp->base_data0);
138  }
139  else if(temp->msg_name == SOUNDS_VOLUME_LEVEL){
140  sounds_volume_level = static_cast<uint8_t>(temp->base_data0);
141  }
142  else if(temp->msg_name == AUDIO_ENABLED){
143  audio_enabled = static_cast<bool>(temp->base_data0);
144  }
145  else if(temp->msg_name == VSYNC_STATE){
146  vsync_flag = static_cast<bool>(temp->base_data0);
147  }
148  else if(temp->msg_name == END_GAME){
149  game_is_running_ = false;
150  }
151  else if(temp->msg_name == SHOW_MOUSE){
152  SDL_ShowCursor(static_cast<bool>(temp->base_data0));
153  }
154  else if(temp->msg_name == EXECUTE_SCRIPT){
155  auto script = static_cast<std::string*>(temp->get_data());
156  gScript_backend.run_script(*script);
157  }
158  else if(temp->msg_name == FULLSCREEN_STATUS){
159  fullscreen_status = static_cast<bool>(temp->base_data0);
160  }
161  else if(temp->msg_name == VIRTUAL_SCREEN_COORDINATES){
162  auto data = temp->base_data0;

166  delete temp;
167  temp = msg_sub.get_next_message();
168  }

175 int8_t game_manager::load_level(const std::string& level_name){
176 
177  //check if it is already loaded.
178  for(auto& i: levels) {
179  if (i.get_name() == level_name) {
180  return 0;
181  }
182  }
183 
184  //otherwise - create it
185  auto temp = ST::level(level_name, &gMessage_bus);
186  if(temp.load() != 0){
187  gMessage_bus.send_msg(new message(LOG_ERROR, make_data<std::string>("Level with name " + level_name + " could not be loaded!")));
188  return -1;
189  }
190  levels.emplace_back(temp);
191 
192  //current level pointer must be reset, because apparently adding to the vector changes the pointer address
193  //(makes sense as it has to reallocate the whole thing)
194  for (auto &level : levels) {
195  if(level.get_name() == active_level){
196  current_level_pointer = &level;
197  break;
198  }
199  }
200  return 0;
201 }

245 void game_manager::start_level(const std::string& level_name) {
246 
247  //set the current level pointer
248  bool is_loaded = false;
249  for (auto &level : levels) {
250  if(level.get_name() == level_name) {
251  current_level_pointer = &level;
252  is_loaded = true;
253  break;
254  }
255  }
256  //if the level wasn't loaded in advance, load it now
257  if(!is_loaded){
258  if(load_level(level_name) == 0) {
259  for (auto &level : levels) {
260  if (level.get_name() == level_name) {
261  current_level_pointer = &level;
262  break;
263  }
264  }
265  } else {
266  gMessage_bus.send_msg(new message(LOG_ERROR, make_data<std::string>("Error starting level " + level_name)));
267  return;
268  }
269  }
270 
271  gScript_backend.close();
272  gScript_backend.initialize(&gMessage_bus, this);
273  active_level = level_name;
274 
275  get_level()->lights.clear();
276  get_level()->entities.clear();
277  get_level()->text_objects.clear();
278 
279  //construct level
280  get_level()->camera.x = 0;
281  get_level()->camera.y = 0;
282 
283  std::string temp = "levels/";
284  temp = temp + active_level;
285  temp = temp + "/level.lua";
286  gScript_backend.run_file(temp);
287 }
