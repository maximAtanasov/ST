81 void game_manager::handle_messages(){
82  auto temp = msg_sub.get_next_message();
83  while(temp != nullptr){
84  switch (temp->msg_name) {
85  case LOAD_LEVEL:
86  load_level(*static_cast<std::string*>(temp->get_data()));
87  break;
88  case RELOAD_LEVEL:
89  reload_level(*static_cast<std::string*>(temp->get_data()));
90  break;
91  case START_LEVEL:
92  start_level(*static_cast<std::string*>(temp->get_data()));
93  break;
94  case UNLOAD_LEVEL:
95  unload_level(*static_cast<std::string*>(temp->get_data()));
96  break;
97  case KEY_PRESSED: {
98  uint8_t key_index = temp->base_data0;
99  keys_pressed_data[key_index] = true;
100  keys_held_data[key_index] = false;
101  keys_released_data[key_index] = false;
102  break;
103  }
104  case KEY_HELD: {
105  uint8_t key_index = temp->base_data0;
106  keys_pressed_data[key_index] = false;
107  keys_held_data[key_index] = true;
108  keys_released_data[key_index] = false;
109  break;
110  }
111  case KEY_RELEASED: {
112  uint8_t key_index = temp->base_data0;
113  keys_pressed_data[key_index] = false;
114  keys_held_data[key_index] = false;
115  keys_released_data[key_index] = true;
116  break;
117  }
118  case MOUSE_X:
119  mouse_x = static_cast<int32_t>(temp->base_data0);
120  break;
121  case MOUSE_Y:
122  mouse_y = static_cast<int32_t>(temp->base_data0);
123  break;
124  case LEFT_TRIGGER:
125  left_trigger = static_cast<int16_t>(temp->base_data0);
126  break;
127  case RIGHT_TRIGGER:
128  right_trigger = static_cast<int16_t>(temp->base_data0);
129  break;
130  case LEFT_STICK_VERTICAL:
131  left_stick_vertical = static_cast<int16_t>(temp->base_data0);
132  break;
133  case LEFT_STICK_HORIZONTAL:
134  left_stick_horizontal = static_cast<int16_t>(temp->base_data0);
135  break;
136  case RIGHT_STICK_VERTICAL:
137  right_stick_vertical = static_cast<int16_t>(temp->base_data0);
138  break;
139  case RIGHT_STICK_HORIZONTAL:
140  right_stick_horizontal = static_cast<int16_t>(temp->base_data0);
141  break;
142  case MUSIC_VOLUME_LEVEL:
143  music_volume_level = static_cast<uint8_t>(temp->base_data0);
144  break;
145  case SOUNDS_VOLUME_LEVEL:
146  sounds_volume_level = static_cast<uint8_t>(temp->base_data0);
147  break;
148  case AUDIO_ENABLED:
149  audio_enabled = static_cast<bool>(temp->base_data0);
150  break;
151  case VSYNC_STATE:
152  vsync_flag = static_cast<bool>(temp->base_data0);
153  break;
154  case END_GAME:
155  game_is_running_ = false;
156  break;
157  case SHOW_MOUSE:
158  SDL_ShowCursor(static_cast<bool>(temp->base_data0));
159  break;
160  case EXECUTE_SCRIPT: {
161  auto script = static_cast<std::string*>(temp->get_data());
162  gScript_backend.run_script(*script);
163  break;
164  }
165  case FULLSCREEN_STATUS:
166  fullscreen_status = static_cast<bool>(temp->base_data0);
167  break;
168  case VIRTUAL_SCREEN_COORDINATES: {
169  auto data = temp->base_data0;
170  v_width = data & 0x0000ffffU;
171  v_height = (data >> 16U) & 0x0000ffffU;
172  break;
173  }
174  case REAL_SCREEN_COORDINATES: {
175  auto data = temp->base_data0;
176  w_width = data & 0x0000ffffU;
177  w_height = (data >> 16U) & 0x0000ffffU;
178  break;
179  }
180  }
181  delete temp;
182  temp = msg_sub.get_next_message();
183  }
184 }

190 int8_t game_manager::load_level(const std::string& level_name){
191 
192  //check if it is already loaded.
193  for(auto& i: levels) {
194  if (i.get_name() == level_name) {
195  return 0;
196  }
197  }
198 
199  //otherwise - create it
200  auto temp = ST::level(level_name, &gMessage_bus);
201  if(temp.load() != 0){
202  gMessage_bus.send_msg(new message(LOG_ERROR, make_data<std::string>("Level with name " + level_name + " could not be loaded!")));
203  return -1;
204  }
205  levels.emplace_back(temp);
206 
207  //current level pointer must be reset, because apparently adding to the vector changes the pointer address
208  //(makes sense as it has to reallocate the whole thing)
209  for (auto &level : levels) {
210  if(level.get_name() == active_level){
211  current_level_pointer = &level;
212  break;
213  }
214  }
215  return 0;
216 }

222 void game_manager::unload_level(const std::string& level_name){
223  for(uint64_t i = 0; i < levels.size(); ++i) {
224  if (levels[i].get_name() == level_name) {
225  levels[i].unload();
226  levels.erase(levels.begin() + i);
227  break;
228  }
229  }
230 
231  //current level pointer must be reset, because apparently adding/removing to/from the vector changes the pointer address
232  //(makes sense as it has to reallocate the whole thing)
233  for (auto &level : levels) {
234  if(level.get_name() == active_level) {
235  current_level_pointer = &level;
236  break;
237  }
238  }

260 void game_manager::start_level(const std::string& level_name) {
261 
262  //set the current level pointer
263  bool is_loaded = false;
264  for (auto &level : levels) {
265  if(level.get_name() == level_name) {
266  current_level_pointer = &level;
267  is_loaded = true;
268  break;
269  }
270  }
271  //if the level wasn't loaded in advance, load it now
272  if(!is_loaded){
273  if(load_level(level_name) == 0) {
274  for (auto &level : levels) {
275  if (level.get_name() == level_name) {
276  current_level_pointer = &level;
277  break;
278  }
279  }
280  } else {
281  gMessage_bus.send_msg(new message(LOG_ERROR, make_data<std::string>("Error starting level " + level_name)));
282  return;
283  }
284  }
285 
286  gScript_backend.close();
287  gScript_backend.initialize(&gMessage_bus, this);
288  active_level = level_name;
289 
290  current_level_pointer->lights.clear();
291  current_level_pointer->entities.clear();
292  current_level_pointer->text_objects.clear();
293 
294  //construct level
295  current_level_pointer->camera.x = 0;
296  current_level_pointer->camera.y = 0;
297  current_level_pointer->camera.limitX2 = v_width;
298  current_level_pointer->camera.limitY2 = v_height;
299 
300  std::string temp = "levels/";
301  temp = temp + active_level;
302  temp = temp + "/level.lua";
303  gScript_backend.run_file(temp);
304 }
