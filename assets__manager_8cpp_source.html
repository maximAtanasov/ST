49 void assets_manager::handle_messages(){
50  message* temp = msg_sub.get_next_message();
51  while(temp != nullptr){
52  if(temp->msg_name == LOAD_LIST){
53  auto path = *static_cast<std::string*>(temp->get_data());
54  load_assets_from_list(path);
55  }
56  else if(temp->msg_name == UNLOAD_LIST){
57  auto path = *static_cast<std::string*>(temp->get_data());
58  unload_assets_from_list(path);
59  }
60  else if(temp->msg_name == LOAD_ASSET){
61  auto path = *static_cast<std::string*>(temp->get_data());
62  load_asset(path);
63  }
64  else if(temp->msg_name == UNLOAD_ASSET){
65  auto path = *static_cast<std::string*>(temp->get_data());
66  unload_asset(path);
67  }
68  else if(temp->msg_name == LOAD_BINARY){
69  auto path = *static_cast<std::string*>(temp->get_data());
70  load_assets_from_binary(path);
71  }
72  destroy_msg(temp);
73  temp = msg_sub.get_next_message();
74  }
75 }

83 int8_t assets_manager::load_assets_from_binary(const std::string& path) {
84  ST::assets_named* assets1 = ST::unpack_binary(path);
85  if(assets1 != nullptr){
86  for(const auto &surface : assets1->surfaces){
87  if(count[surface.first] > 0){
88  count[surface.first]++;
89  continue;
90  }else{
91  gMessage_bus->send_msg(make_msg(LOG_SUCCESS, make_data<std::string>("Unpacking " + surface.first)));
92  count[surface.first]++;
93  std::hash<std::string> hash_f;
94  size_t hashed = hash_f(surface.first);
95  all_assets.surfaces[hashed] = surface.second;
96  }
97  }
98  for(const auto &chunk : assets1->chunks){
99  if(count[chunk.first] > 0){
100  count[chunk.first]++;
101  continue;
102  }else{
103  gMessage_bus->send_msg(make_msg(LOG_SUCCESS, make_data<std::string>("Unpacking " + chunk.first)));
104  count[chunk.first]++;
105  std::hash<std::string> hash_f;
106  size_t hashed = hash_f(chunk.first);
107  all_assets.chunks[hashed] = chunk.second;
108  }
109  }
110  for(const auto &music : assets1->music){
111  if(count[music.first] > 0){
112  count[music.first]++;
113  continue;
114  }else{
115  gMessage_bus->send_msg(make_msg(LOG_SUCCESS, make_data<std::string>("Unpacking " + music.first)));
116  count[music.first]++;
117  std::hash<std::string> hash_f;
118  size_t hashed = hash_f(music.first);
119  all_assets.music[hashed] = music.second;
120  }
121  }
122  delete assets1;
123  }else{
124  return -1;
125  }

163 int8_t assets_manager::load_asset(std::string path){
164  //ignore a comment
165  if(path.at(0) == '#'){
166  return 0;
167  }
168 
169  //check if the file is already loaded and increase the reference count
170  if(count[path] > 0){
171  count[path]++;
172  return 0;
173  }
174 
175  std::string extention;
176  extention = ST::get_file_extension(path);
177  if(extention == "bin"){
178  gMessage_bus->send_msg(make_msg(LOG_INFO, make_data<std::string>("Loading from binary " + path)));
179  }else {
180  gMessage_bus->send_msg(make_msg(LOG_INFO, make_data<std::string>("Loading " + path)));
181  }
182  std::hash<std::string> hash_f;
183 
184  //Handle the different extentions - currently png, wav, mp3, ttf
185  if(extention == "png" || extention == "webp"){
186  SDL_Surface* temp1 = IMG_Load(path.c_str());
187  if(temp1 != nullptr){
188  path = trim_path(path);
189  size_t string_hash = hash_f(path);
190  all_assets.surfaces[string_hash] = temp1;
191  count[path]++;
192  }else{
193  gMessage_bus->send_msg(make_msg(LOG_ERROR, make_data<std::string>("File " + path + " not found")));
194  return -1;
195  }
196  }else if(extention == "wav"){
197  Mix_Chunk* temp1 = Mix_LoadWAV(path.c_str());
198  if (temp1 != nullptr){
199  for(Uint32 i = 0; i < path.size(); i++) {
200  if (path.at(i) == '/') {
201  path.erase(0, i+1);
202  }
203  }
204  size_t string_hash = hash_f(path);
205  all_assets.chunks[string_hash] = temp1;
206  count[path]++;
207  }
208  else{
209  gMessage_bus->send_msg(make_msg(LOG_ERROR, make_data<std::string>("File " + path + " not found")));
210  return -1;
211  }
212  }else if(extention == "ogg"){
213  Mix_Music* temp1 = Mix_LoadMUS(path.c_str());
214  if (temp1 != nullptr) {
215  for(Uint32 i = 0; i < path.size(); i++) {
216  if (path.at(i) == '/') {
217  path.erase(0, i+1);
218  }
219  }
220  size_t string_hash = hash_f(path);
221  all_assets.music[string_hash] = temp1;
222  count[path]++;
223  }
224  else {
225  gMessage_bus->send_msg(make_msg(LOG_ERROR, make_data<std::string>("File " + path + " not found")));
226  return -1;
227  }
228  }else if(extention == "bin"){
229  load_assets_from_binary(path);
230  }else{ //if file is a font
231  std::vector<std::string> result;
232  std::istringstream iss(path);
233  for(std::string path_; iss >> path;)
234  result.push_back(path);
235  int size;
236  try{
237  std::stringstream convert(result.at(1));
238  convert >> size;
239  }catch(const std::out_of_range& e){
240  (void)e;
241  gMessage_bus->send_msg(make_msg(LOG_ERROR, make_data<std::string>("Font size not specified!")));
242  return -1;
243  }
244  std::string font = result.at(0);
245  TTF_Font* tempFont = TTF_OpenFont(font.c_str(), size);
246  if(tempFont != nullptr){
247  for(Uint32 i = 0; i < font.size(); i++) {
248  if (font.at(i) == '/') {
249  font.erase(0, i+1);
250  }
251  }
252  std::string font_and_size = font + result.at(1);
253  all_assets.fonts[font_and_size] = tempFont;
254  count[font_and_size]++;
255  }else{
256  gMessage_bus->send_msg(make_msg(LOG_ERROR, make_data<std::string>("File " + font + " not found!")));
257  return -1;
258  }
259  }
260  return 0;
261 }

270  file.open(path.c_str());

326 int8_t assets_manager::unload_asset(std::string path){
327  //Ignore comments
328  if(path.at(0) == '#'){
329  return 0;
330  }
331  if(count[path] > 1){
332  count[path]--;
333  return 0;
334  }
335  std::string extention = ST::get_file_extension(path);
336  gMessage_bus->send_msg(make_msg(LOG_INFO, make_data<std::string>("Unloading " + path)));
337  if(extention == "png" || extention == "webp"){
338  std::hash<std::string> hash_f;
339  size_t string_hash = hash_f(path);
340  SDL_FreeSurface(all_assets.surfaces[string_hash]);
341  all_assets.surfaces[string_hash] = nullptr;
342  count[path]--;
343  }else if(extention == "wav"){
344  std::hash<std::string> hash_f;
345  size_t string_hash = hash_f(path);
346  Mix_FreeChunk(all_assets.chunks[string_hash]);
347  count[path]--;
348  }else if(extention == "ogg"){
349  std::hash<std::string> hash_f;
350  size_t string_hash = hash_f(path);
351  Mix_FreeMusic(all_assets.music[string_hash]);
352  count[path]--;
353  }else if(extention == "bin"){
354  return unload_assets_from_binary(path);
355  }else{ //if file is a font
356  TTF_CloseFont(all_assets.fonts[path]);
357  count[path]--;
358  }
359  return 0;
360 }
