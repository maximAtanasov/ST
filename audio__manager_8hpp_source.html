119 inline void audio_manager::play_sound(uint16_t arg, uint8_t volume, int8_t loops) const{
120  auto data = chunks_ptr->find(arg);
121  if(data != chunks_ptr->end()){
122  if(!muted){
123  Mix_VolumeChunk(data->second, static_cast<int>(static_cast<float>(volume) / chunk_playback_volume_ratio));
124  }
125  if(Mix_PlayChannel( -1, data->second, loops ) == -1){
126  gMessage_bus.send_msg(new message(LOG_ERROR, make_data<std::string>("Mix_PlayChannel Error " + std::string(Mix_GetError()))));
127  }
128  }
129 }

137 inline void audio_manager::play_music(uint16_t arg, uint8_t volume, int8_t loops) const{
