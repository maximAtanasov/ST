17 // These these are purely visual - you have to watch them run and
18 // look out for anything wrong. This means that you won't get
19 // a test failure if for some reason nothing is rendered.
20 // However these are still useful as they provide you with a quick
21 // way to test if any of the specific rendering functions is broken.

32  int16_t test_window_height = 0;

34  void SetUp() override{
35  initialize_SDL();
36  SDL_DisplayMode DM{};
37  SDL_GetDisplayMode(0, 0, &DM);
38  test_window_width = static_cast<int16_t>(DM.w);
39  test_window_height = static_cast<int16_t>(DM.h);
40 
41  test_window = SDL_CreateWindow("TEST", 0, 0, test_window_width, test_window_height, 0);
42 
43  ST::renderer_sdl::initialize(test_window, test_window_width, test_window_height);
44 
45  SDL_SetWindowFullscreen(test_window, fullscreen);
46  SDL_ShowCursor(0);
47 
48  ST::renderer_sdl::clear_screen({0,0,0,0});
49  }
50 
51  void TearDown() override{
52  ST::renderer_sdl::close();
53  SDL_DestroyWindow(test_window);
54  close_SDL();
55  }
