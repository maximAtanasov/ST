80 void window_manager::handle_messages(){
81  message* temp = msg_sub.get_next_message();
82  while(temp != nullptr){
83  switch (temp->msg_name) {
84  case SET_FULLSCREEN: {
85  auto arg = static_cast<bool>(temp->base_data0);
86  set_fullscreen(arg);
87  gMessage_bus.send_msg(new message(FULLSCREEN_STATUS, arg));
88  break;
89  } case SET_WINDOW_BRIGHTNESS: {
90  auto arg = *static_cast<float*>(temp->get_data());
91  set_brightness(arg);
92  gMessage_bus.send_msg(new message(LOG_SUCCESS, make_data<std::string>("Brightness set to: " + std::to_string(arg))));
93  break;
94  } case SET_WINDOW_RESOLUTION: {
95  auto data = temp->base_data0;
96  int16_t new_width = data & 0x0000ffffU;
97  int16_t new_height = (data >> 16U) & 0x0000ffffU;
98  if(new_width != width && new_height != height) {
99  uint32_t screen_width_height = new_width | (new_height << 16U); // NOLINT(hicpp-signed-bitwise)
100  gMessage_bus.send_msg(new message(REAL_SCREEN_COORDINATES, screen_width_height));
101  SDL_SetWindowSize(window, new_width, new_height);
102  width = new_width;
103  height = new_height;
104  SDL_SetWindowPosition(window, SDL_WINDOWPOS_CENTERED, SDL_WINDOWPOS_CENTERED); // NOLINT(hicpp-signed-bitwise)
105  }
106  break;
107  }
108  }
109  delete temp;
110  temp = msg_sub.get_next_message();
111  }
112 }

126 void window_manager::set_fullscreen(bool arg){
127 #ifdef _MSC_VER

129  SDL_SetWindowFullscreen(window, SDL_WINDOW_FULLSCREEN_DESKTOP);
130  SDL_SetWindowFullscreen(window, SDL_WINDOW_FULLSCREEN);
131  }
132  else if(!arg && (SDL_GetWindowFlags(window) & SDL_WINDOW_FULLSCREEN)){ // NOLINT(hicpp-signed-bitwise)
133  SDL_SetWindowFullscreen(window, SDL_WINDOW_FULLSCREEN_DESKTOP);
134  SDL_SetWindowFullscreen(window, 0);
135  SDL_SetWindowFullscreen(window, SDL_WINDOW_FULLSCREEN_DESKTOP);
136  SDL_SetWindowFullscreen(window, 0);
137  SDL_SetWindowPosition(window, SDL_WINDOWPOS_CENTERED, SDL_WINDOWPOS_CENTERED); // NOLINT(hicpp-signed-bitwise)
138  }
139 #else
140  if(arg && !(SDL_GetWindowFlags(window) & SDL_WINDOW_FULLSCREEN)) {
141  SDL_SetWindowFullscreen(window, SDL_WINDOW_FULLSCREEN);
142  }
143  else if(!arg && (SDL_GetWindowFlags(window) & SDL_WINDOW_FULLSCREEN)){
144  SDL_SetWindowFullscreen(window, SDL_WINDOW_FULLSCREEN_DESKTOP);
145  SDL_SetWindowFullscreen(window, 0);
