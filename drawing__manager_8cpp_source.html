59 void drawing_manager::update(const ST::level& temp, double fps, const console& cnsl){
60  Camera = temp.Camera;
61  handle_messages();
62 
63  ticks = SDL_GetTicks(); //CPU ticks since start
64  ST::renderer_sdl::clear_screen();
65  ST::renderer_sdl::draw_background(temp.background);
66  draw_entities(temp.entities);
67  ST::renderer_sdl::draw_overlay(temp.overlay, ticks % temp.overlay_spriteNum, temp.overlay_spriteNum);
68  draw_text_objects(temp.text_objects);
69  //draw the lights when we are sure they are processed
70  if(lighting_enabled) {
71  process_lights(temp.lights);
72  draw_lights();
73  }
74  //Draw debug info and the console in a debug build
75  if (collisions_shown) {
76  draw_collisions(temp.entities);
77  draw_coordinates(temp.entities);
78  }
79  draw_fps(fps);
80  draw_console(cnsl);
81 
82  ST::renderer_sdl::present();
83 }

143 void drawing_manager::process_lights(const std::vector<ST::light>& lights){
144  if(!lighting_enabled){
145  return;
146  }
147  for(int i = 0; i < w_width; i++){
148  for(int j = 0; j < w_height; j++){
149  lightmap[i][j] = darkness_level;
150  }
151  }
152  for (const auto &light : lights) {
153  int x = light.origin_x, y = light.origin_y;
154  if(!light.is_static){
155  x -= Camera.x;
156  y -= Camera.y;
157  }
158  double count = 0;
159  double step = darkness_level/ static_cast<double>(light.radius);
160  count = 0;
161  int radius = light.radius;
162  int intensity = light.intensity;
163  if(x - radius - intensity > w_width || y - radius - intensity > w_height)
164  continue;
165  double step2 = 0;
166  for(int i = y; i < y + radius + intensity; i++){
167  for(int j = x; j < x + radius + intensity; j++){
168  if(j > 0 && j < w_width && i > 0 && i < w_height)
169  lightmap[j][i] = (uint8_t) light.brightness + (uint8_t)count;
170  if(count + light.brightness < darkness_level && j > x + intensity)
171  count += step;
172  }
173  count = 0;
174  count += step2;
175  if(step2 + light.brightness < darkness_level && i > y + intensity)
176  step2 += step;
177  }
178  count = 0;
179  step2 = 0;
180  for(int i = y; i > y - radius - intensity; i--){
181  for(int j = x; j > x - radius - intensity; j--){
182  if(j > 0 && j < w_width && i > 0 && i < w_height)
183  lightmap[j][i] = light.brightness + (uint8_t)count;
184  if(count + light.brightness < darkness_level && j < x - intensity)
185  count += step;
186  }
187  count = 0;
188  count += step2;
189  if(step2 + light.brightness < darkness_level && i < y - intensity)
190  step2 += step;
191  }
192  count = 0;
193  step2 = 0;
194  for(int i = y; i > y - radius - intensity; i--){
195  for(int j = x; j < x + radius + intensity; j++){
196  if(j > 0 && j < w_width && i > 0 && i < w_height)
197  lightmap[j][i] = light.brightness + (uint8_t)count;
198  if(count + light.brightness < darkness_level && j > x + intensity)
199  count += step;
200  }
201  count = 0;
202  count += step2;
203  if(step2 + light.brightness < darkness_level && i < y - intensity)
204  step2 += step;
205  }
206  count = 0;
207  step2 = 0;
208  for(int i = y; i < y + radius + intensity; i++){
209  for(int j = x; j > x - radius - intensity; j--){
210  if(j > 0 && j < w_width && i > 0 && i < w_height)
211  lightmap[j][i] = light.brightness + (uint8_t)count;
212  if(count + light.brightness< darkness_level && j < x - intensity)
213  count += step;
214  }
215  count = 0;
216  count += step2;
217  if(step2 + light.brightness < darkness_level && i > y + intensity) {
218  step2 += step;
219  }
220  }
221  }

235  tempRect.y = j - a*lights_quality;

258 void drawing_manager::handle_messages(){
259  message* temp = msg_sub.get_next_message();
260  while(temp != nullptr){
261  if(temp->msg_name == SET_VSYNC){
262  auto arg = static_cast<bool*>(temp->get_data());
263  if(*arg){
264  ST::renderer_sdl::vsync_on();
265  }else{
266  ST::renderer_sdl::vsync_off();
267  }
268  gMessage_bus->send_msg(make_msg(VSYNC_STATE, make_data<>(*arg)));
269  }
270  else if(temp->msg_name == SET_DARKNESS){
271  set_darkness(*(uint8_t*)temp->get_data());
272  }
273  else if(temp->msg_name == SHOW_COLLISIONS){
274  auto arg = static_cast<bool*>(temp->get_data());
275  if(*arg) {
276  show_collisions();
277  }
278  else if(!(*arg)){
279  hide_collisions();
280  }
281  }
282  else if(temp->msg_name == SHOW_FPS){
283  auto arg = static_cast<bool*>(temp->get_data());
284  if(*arg) {
285  show_fps = true;
286  }
287  else if(!(*arg)){
288  show_fps = false;
289  }
290  }
291  else if(temp->msg_name == ENABLE_LIGHTING){
292  auto arg = static_cast<bool*>(temp->get_data());
293  if(*arg) {
294  lighting_enabled = true;
295  }
296  else if(!(*arg)){
297  lighting_enabled = false;
298  }
299  }
300  else if(temp->msg_name == SURFACES_ASSETS) {
301  auto surfaces = *static_cast<ska::bytell_hash_map<size_t, SDL_Surface *>**>(temp->get_data());
302  ST::renderer_sdl::upload_surfaces(surfaces);
303  }
304  else if(temp->msg_name == FONTS_ASSETS) {
305  auto fonts = *static_cast<ska::bytell_hash_map<std::string, TTF_Font *>**>(temp->get_data());
306  ST::renderer_sdl::upload_fonts(fonts);
307  }
308  else if(temp->msg_name == SET_INTERNAL_RESOLUTION) {
309  auto res = *static_cast<std::tuple<int16_t, int16_t>*>(temp->get_data());
310  gMessage_bus->send_msg(make_msg(VIRTUAL_SCREEN_COORDINATES, make_data(res)));
311  ST::renderer_sdl::set_resolution(std::get<0>(res), std::get<1>(res));
312  w_width = std::get<0>(res);
313  w_height = std::get<1>(res);
314  }
315  destroy_msg(temp);
316  temp = msg_sub.get_next_message();
317  }
318 }
