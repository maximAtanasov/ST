36 void physics_manager::process_horizontal(std::vector<ST::entity>* entities, int8_t friction) {
37  for(uint64_t k = 0; k < entities->size(); k++) {
38  auto& entity = entities->operator[](k);
39  //handle horizontal velocity
40  if (entity.is_affected_by_physics()) {
41  if (entity.velocity_x > 0) {
42  for (int j = 0; j < entity.velocity_x; j++) {
43  if (entity_set_x(entity.x + 1, k, entities) == 0) {
44  entity.velocity_x = 0;
45  break;
46  }
47  }
48  for (int j = 0; j < friction && entity.velocity_x > 0; j++) {
49  (entity.velocity_x = (static_cast<int8_t>(entity.velocity_x - 1)));
50  }
51  } else if (entity.velocity_x < 0) {
52  for (int j = 0; j > entity.velocity_x; j--) {
53  if (entity_set_x(entity.x - 1, k, entities) == 0) {
54  entity.velocity_x = 0;
55  break;
56  }
57  }
58  for (int j = 0; j < friction && entity.velocity_x < 0; j++) {
59  (entity.velocity_x = (static_cast<int8_t>(entity.velocity_x + 1)));
60  }
61  }
62  }
63  }

73 void physics_manager::process_vertical(std::vector<ST::entity>* entities, int8_t gravity, int32_t level_floor) {
74  for(uint64_t k = 0; k < entities->size(); k++) {
75  auto& entity = entities->operator[](k);
76  if (entity.is_affected_by_physics()) {
77  //handle vertical velocity
78  const int8_t objectVelocity = entity.velocity_y + gravity;
79  if (objectVelocity < 0) {
80  for (int j = 0; j > objectVelocity; j--) {
81  if (entity_set_y(entity.y - 1, k, entities) == 0) {
82  break;
83  }
84  }
85  } else if (objectVelocity > 0) {
86  for (int j = 0; j < objectVelocity; j++) {
87  if (entity.y + entity.get_col_y_offset() < level_floor) {
88  if (entity_set_y(entity.y + 1, k, entities) == 0) {
89  break;
90  }
91  }
92  }
93  }
94  //decrease velocity of objects (apply gravity really)
95  int8_t realVelocity = objectVelocity - gravity;
96  if (realVelocity < 0) {
97  entity.velocity_y = (static_cast<int8_t>(realVelocity + 2));
98  }
99  }
100  }

107 void physics_manager::handle_messages(){
108  message* temp = msg_sub.get_next_message();
109  while(temp != nullptr){
110  if(temp->msg_name == SET_GRAVITY){
111  gravity = static_cast<int8_t>(temp->base_data0);
112  }else if(temp->msg_name == SET_FLOOR){
113  level_floor = static_cast<int32_t>(temp->base_data0);
114  }else if(temp->msg_name == SET_FRICTION){
115  friction = static_cast<int8_t>(temp->base_data0);
116  }else if(temp->msg_name == PAUSE_PHYSICS){
117  physics_paused = true;
118  }else if(temp->msg_name == UNPAUSE_PHYSICS){
119  physics_paused = false;
120  }
121  delete temp;
122  temp = msg_sub.get_next_message();
123  }
124 }
