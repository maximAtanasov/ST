131 uint16_t ST::renderer_sdl::draw_text_lru_cached(uint16_t font, const std::string& arg2, int x, int y, SDL_Color color_font){
132  TTF_Font* _font = fonts[font];
133  int32_t texW = 0;
134  if(_font != nullptr) [[likely]] {
135  int32_t texH;
136  SDL_Texture* cached_texture = font_cache::get_cached_string(arg2, font);
137  if(cached_texture != nullptr){ //if the given string (with same size and font) is already cached, get it from cache
138  SDL_QueryTexture(cached_texture, nullptr, nullptr, &texW, &texH);
139  SDL_Rect Rect = {x, y - texH, texW, texH};
140  SDL_SetTextureColorMod(cached_texture, color_font.r, color_font.g, color_font.b);
141  SDL_RenderCopy(sdl_renderer, cached_texture, nullptr, &Rect);
142  }else{ //else create a texture, render it, and then cache it - this is costly, so pick a good cache size
143  SDL_Surface* text = TTF_RenderUTF8_Blended(_font, arg2.c_str(), color_font);
144  SDL_Texture* texture = SDL_CreateTextureFromSurface(sdl_renderer, text);
145  SDL_QueryTexture(texture, nullptr, nullptr, &texW, &texH);
146  SDL_Rect Rect = {x, y - texH, texW, texH};
147  SDL_SetTextureColorMod(texture, color_font.r, color_font.g, color_font.b);
148  SDL_RenderCopy(sdl_renderer, texture, nullptr, &Rect);
149  SDL_FreeSurface(text);
150  font_cache::cache_string(arg2, texture, font);
151  }
152  }
153  return static_cast<uint16_t>(texW);
154 }

169 uint16_t ST::renderer_sdl::draw_text_cached_glyphs(uint16_t font, const std::string& arg2, const int x, const int y, const SDL_Color color_font) {

228 void ST::renderer_sdl::process_surfaces(std::vector<std::pair<uint16_t, SDL_Surface *>> &surfaces_pairs) {
229  if(surfaces_pairs.empty()) {
230  return;
231  } else if(surfaces_pairs.size() == 1) { //Only one surface in vector => no point in creating an atlas.
232  auto tex = ST::renderer_sdl::texture{};
233  tex.width = surfaces_pairs.back().second->w;
234  tex.height = surfaces_pairs.back().second->h;
235  tex.atlas_v_offset = 0;
236  tex.atlas_h_offset = 0;
237  tex.atlas = SDL_CreateTextureFromSurface(sdl_renderer, surfaces_pairs.back().second);
238  textures[surfaces_pairs.back().first] = tex;
239  return;
240  }
241  sort(surfaces_pairs.begin(), surfaces_pairs.end(), sort_by_surface_width_height);
242 
243  std::vector<std::pair<uint16_t, SDL_Surface*>> remaining_surface_pairs{};
244  uint8_t atlas[atlas_grid_size][atlas_grid_size] = {0};
245  int h_index = 0;
246  int v_index = 0;
247 
248  SDL_Texture* atlas_texture = SDL_CreateTexture(sdl_renderer, SDL_PIXELFORMAT_RGBA32, SDL_TEXTUREACCESS_STATIC, ATLAS_SIZE, ATLAS_SIZE);
249  SDL_SetTextureBlendMode(atlas_texture, SDL_BLENDMODE_BLEND);
250  for(auto& it : surfaces_pairs) {
251  int t_width = it.second->w/32;
252  int t_height = it.second->h/32;
253 
254  //Texture either too large for atlas or has dimensions not power of 2
255  if((it.second->w & (it.second->w - 1)) != 0 || (it.second->h & (it.second->h - 1)) != 0 || t_width > atlas_grid_size || t_height > atlas_grid_size) {
256  auto tex = ST::renderer_sdl::texture{};
257  tex.width = it.second->w;
258  tex.height = it.second->h;
259  tex.atlas_v_offset = 0;
260  tex.atlas_h_offset = 0;
261  tex.atlas = SDL_CreateTextureFromSurface(sdl_renderer, it.second);
262  textures[it.first] = tex;
263  continue;
264  }
265 
266  while(atlas[h_index][v_index] != 0) {
267  ++h_index;
268  if(h_index == atlas_grid_size) {
269  h_index = 0;
270  ++v_index;
271 
272  if(v_index == atlas_grid_size) {
273  remaining_surface_pairs.emplace_back(it);
274  break;
275  }
276  }
277  }
278 
279  if(v_index + t_height > atlas_grid_size) {
280  remaining_surface_pairs.emplace_back(it);
281  continue;
282  }
283 
284  for(int i = h_index; i < h_index + t_width; ++i) {
285  for(int j = v_index; j < v_index + t_height; ++j){
286  atlas[i][j] = 1;
287  }
288  }
289 
290  //Blit to the larger surface
291  auto tex = ST::renderer_sdl::texture{};
292  tex.width = it.second->w;
293  tex.height = it.second->h;
294  tex.atlas_h_offset = h_index * 32;
295  tex.atlas_v_offset = v_index * 32;
296  tex.atlas = atlas_texture;
297  textures[it.first] = tex;
298 
299  SDL_Rect dst_rect = {tex.atlas_h_offset, tex.atlas_v_offset, tex.width, tex.height};
300  SDL_UpdateTexture(atlas_texture, &dst_rect, it.second->pixels, it.second->pitch);
301 
302  h_index += t_width;
303  if(h_index == atlas_grid_size) {
304  h_index = 0;
305  v_index += 1;
306  }
307  }
308  ST::renderer_sdl::process_surfaces(remaining_surface_pairs);
309 }

346 void ST::renderer_sdl::cache_font(TTF_Font* Font, uint16_t font_and_size){
347  SDL_Color color_font = {255, 255, 255, 255};
348  char temp[2];
349  temp[1] = 0;
350  std::vector<SDL_Texture*> tempVector;
351  for(char j = 32; j < 127; j++) {
352  temp[0] = j;
353  SDL_Surface* glyph = TTF_RenderUTF8_Blended(Font, temp, color_font);
354  SDL_Texture* texture = SDL_CreateTextureFromSurface(sdl_renderer, glyph);
355  tempVector.push_back(texture);
356  SDL_FreeSurface(glyph);
357  }
358  fonts_cache[font_and_size] = tempVector;
