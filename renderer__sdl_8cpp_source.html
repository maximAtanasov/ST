133 uint16_t ST::renderer_sdl::draw_text_lru_cached(uint16_t font, const std::string& arg2, int x, int y, SDL_Color color_font){
134  TTF_Font* _font = fonts[font];
135  int32_t texW = 0;
136  if(_font != nullptr) [[likely]] {
137  int32_t texH;
138  SDL_Texture* cached_texture = font_cache::get_cached_string(arg2, font);
139  if(cached_texture != nullptr){ //if the given string (with same size and font) is already cached, get it from cache
140  SDL_QueryTexture(cached_texture, nullptr, nullptr, &texW, &texH);
141  SDL_Rect Rect = {x, y - texH, texW, texH};
142  SDL_SetTextureColorMod(cached_texture, color_font.r, color_font.g, color_font.b);
143  SDL_RenderCopy(sdl_renderer, cached_texture, nullptr, &Rect);
144  }else{ //else create a texture, render it, and then cache it - this is costly, so pick a good cache size
145  SDL_Surface* text = TTF_RenderUTF8_Blended(_font, arg2.c_str(), color_font);
146  SDL_Texture* texture = SDL_CreateTextureFromSurface(sdl_renderer, text);
147  SDL_QueryTexture(texture, nullptr, nullptr, &texW, &texH);
148  SDL_Rect Rect = {x, y - texH, texW, texH};
149  SDL_SetTextureColorMod(texture, color_font.r, color_font.g, color_font.b);
150  SDL_RenderCopy(sdl_renderer, texture, nullptr, &Rect);
151  SDL_FreeSurface(text);
152  font_cache::cache_string(arg2, texture, font);
153  }
154  }
155  return static_cast<uint16_t>(texW);
156 }

230 void ST::renderer_sdl::process_surfaces(std::vector<std::pair<uint16_t, SDL_Surface *>> &surfaces_pairs) {
231  if(surfaces_pairs.empty()) {
232  return;
233  } else if(surfaces_pairs.size() == 1) { //Only one surface in vector => no point in creating an atlas.
234  auto tex = ST::renderer_sdl::texture{};
235  tex.width = surfaces_pairs.back().second->w;
236  tex.height = surfaces_pairs.back().second->h;
237  tex.atlas_v_offset = 0;
238  tex.atlas_h_offset = 0;
239  tex.atlas = SDL_CreateTextureFromSurface(sdl_renderer, surfaces_pairs.back().second);
240  textures[surfaces_pairs.back().first] = tex;
241  return;
242  }
243  sort(surfaces_pairs.begin(), surfaces_pairs.end(), sort_by_surface_width_height);
244 
245  std::vector<std::pair<uint16_t, SDL_Surface*>> remaining_surface_pairs{};
246  uint8_t atlas[atlas_grid_size][atlas_grid_size] = {0};
247  int h_index = 0;
248  int v_index = 0;
249 
250  SDL_Texture* atlas_texture = SDL_CreateTexture(sdl_renderer, SDL_PIXELFORMAT_RGBA32, SDL_TEXTUREACCESS_STATIC, ATLAS_SIZE, ATLAS_SIZE);
251  SDL_SetTextureBlendMode(atlas_texture, SDL_BLENDMODE_BLEND);
252  for(auto& it : surfaces_pairs) {
253  int t_width = it.second->w/32;
254  int t_height = it.second->h/32;
255 
256  //Texture either too large for atlas or has dimensions not power of 2
257  if((it.second->w & (it.second->w - 1)) != 0 || (it.second->h & (it.second->h - 1)) != 0 || t_width > atlas_grid_size || t_height > atlas_grid_size) {
258  auto tex = ST::renderer_sdl::texture{};
259  tex.width = it.second->w;
260  tex.height = it.second->h;
261  tex.atlas_v_offset = 0;
262  tex.atlas_h_offset = 0;
263  tex.atlas = SDL_CreateTextureFromSurface(sdl_renderer, it.second);
264  textures[it.first] = tex;
265  continue;
266  }
267 
268  while(atlas[h_index][v_index] != 0) {
269  ++h_index;
270  if(h_index == atlas_grid_size) {
271  h_index = 0;
272  ++v_index;
273 
274  if(v_index == atlas_grid_size) {
275  remaining_surface_pairs.emplace_back(it);
276  break;
277  }
278  }
279  }
280 
281  if(v_index + t_height > atlas_grid_size) {
282  remaining_surface_pairs.emplace_back(it);
283  continue;
284  }
285 
286  for(int i = h_index; i < h_index + t_width; ++i) {
287  for(int j = v_index; j < v_index + t_height; ++j){
288  atlas[i][j] = 1;
289  }
290  }
291 
292  //Blit to the larger surface
293  auto tex = ST::renderer_sdl::texture{};
294  tex.width = it.second->w;
295  tex.height = it.second->h;
296  tex.atlas_h_offset = h_index * 32;
297  tex.atlas_v_offset = v_index * 32;
298  tex.atlas = atlas_texture;
299  textures[it.first] = tex;
300 
301  SDL_Rect dst_rect = {tex.atlas_h_offset, tex.atlas_v_offset, tex.width, tex.height};
302  SDL_UpdateTexture(atlas_texture, &dst_rect, it.second->pixels, it.second->pitch);
303 
304  h_index += t_width;
305  if(h_index == atlas_grid_size) {
306  h_index = 0;
307  v_index += 1;
308  }
309  }
310  ST::renderer_sdl::process_surfaces(remaining_surface_pairs);
311 }

354 void ST::renderer_sdl::cache_font(TTF_Font* Font, uint16_t font_and_size) {
355  SDL_Color color_font = {255, 255, 255, 255};
356  char temp[2];
357  temp[1] = 0;
358  std::vector<ST::renderer_sdl::texture> result_glyphs{};
359  std::vector<SDL_Surface*> glyph_surfaces{};
360  result_glyphs.reserve(95);
361  glyph_surfaces.reserve(95);
362  int32_t total_width = 0;
363  int32_t max_height = 0;
364  for(char j = 32; j < 127; j++) {
365  temp[0] = j;
366  SDL_Surface* glyph_surface = TTF_RenderUTF8_Blended(Font, temp, color_font);
367  total_width += glyph_surface->w;
368  if(glyph_surface->h > max_height) {
369  max_height = glyph_surface->h;
370  }
371  glyph_surfaces.emplace_back(glyph_surface);
372  }
373  SDL_Texture* atlas_texture = SDL_CreateTexture(sdl_renderer, SDL_PIXELFORMAT_ARGB8888, SDL_TEXTUREACCESS_STATIC, total_width, max_height);
374  SDL_SetTextureBlendMode(atlas_texture, SDL_BLENDMODE_BLEND);
375  int atlas_offset = 0;
376  for(auto surface : glyph_surfaces) {
377  ST::renderer_sdl::texture glyph{};
378  glyph.width = surface->w;
379  glyph.height = surface->h;
380  glyph.atlas_h_offset = atlas_offset;
381  glyph.atlas_v_offset = 0;
382  glyph.atlas = atlas_texture;
383  result_glyphs.push_back(glyph);
384  SDL_Rect dst_rect = {glyph.atlas_h_offset, glyph.atlas_v_offset, glyph.width, glyph.height};
385  SDL_UpdateTexture(atlas_texture, &dst_rect, surface->pixels, surface->pitch);
386  SDL_FreeSurface(surface);
387  atlas_offset += glyph.width;
388  }
389  fonts_cache[font_and_size] = result_glyphs;
390 }
