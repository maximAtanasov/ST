59 void audio_manager::handle_messages(){
60  message* temp = msg_sub.get_next_message();
61  while(temp != nullptr){
62  if(temp->msg_name == PLAY_SOUND){
63  auto data = static_cast<std::tuple<size_t, int, int>*>(temp->get_data());
64  size_t name = std::get<0> (*data);
65  int volume = std::get<1> (*data);
66  int loops = std::get<2> (*data);
67  play_sound(name, volume, loops);
68  }
69  else if(temp->msg_name == PLAY_MUSIC){
70  auto data = (std::tuple<size_t, int, int>*)temp->get_data();
71  size_t name = std::get<0> (*data);
72  int volume = std::get<1> (*data);
73  int loops = std::get<2> (*data);
74  play_music(name, volume, loops);
75  }
76  else if(temp->msg_name == STOP_MUSIC){
77  stop_music();
78  log(SUCCESS, "Music stopped");
79  }
80  else if(temp->msg_name == STOP_ALL_SOUNDS){
81  stop_channels();
82  log(SUCCESS, "Sounds stopped");
83  }
84  else if(temp->msg_name == TOGGLE_AUDIO){
85  if(volume > 0){
86  log(SUCCESS, "Audio OFF");
87  mute();
88  set_volume(0);
89  }
90  else{
91  set_volume(MIX_MAX_VOLUME);
92  unmute();
93  log(SUCCESS, "Audio ON");
94  }
95  gMessage_bus->send_msg(make_msg(VOLUME_LEVEL, make_data(volume)));
96  }
97  else if(temp->msg_name == ASSETS){
98  auto temp_ptr = (ST::assets**)temp->get_data();
99  assets_ptr = *temp_ptr;
100  }else if(temp->msg_name == SET_VOLUME){
101  auto arg = static_cast<uint8_t*>(temp->get_data());
102  set_volume(*arg);
103  gMessage_bus->send_msg(make_msg(VOLUME_LEVEL, make_data(volume)));
104  }
105  destroy_msg(temp);
106  temp = msg_sub.get_next_message();
107  }
108 }
